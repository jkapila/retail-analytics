def extract_num(string):
    return int(re.sub('[^0-9,]', "", string))

def to_months(Yr_mon):
    try:
        return sum([extract_num(s)*12 if 'Year' in s else extract_num(s) for s in Yr_mon.split('and')])
    except:
        return np.nan
